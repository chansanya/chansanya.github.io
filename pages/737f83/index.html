修改MySql加密方式

select user,host,plugin from user where user='root';

ALTER USER 'root'@'%' IDENTIFIED BY '1223' PASSWORD EXPIRE NEVER;
#更新用户密码
ALTER USER 'root'@'%' IDENTIFIED WITH mysql_native_password BY '1223';
#刷新权限
FLUSH PRIVILEGES;
